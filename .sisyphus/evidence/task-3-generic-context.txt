export interface ToolExecutionContext {
  readonly requestId: string;
  readonly toolName: string;
  readonly config: RuntimeConfig;
}

### After:
```typescript
import type { BaseRuntimeConfig, RuntimeConfig } from '../contracts/runtime-config.js';

export interface BaseToolExecutionContext<TConfig extends BaseRuntimeConfig = BaseRuntimeConfig> {
  readonly requestId: string;
  readonly toolName: string;
  readonly config: TConfig;
}

export type ToolExecutionContext = BaseToolExecutionContext<RuntimeConfig>;
```

## Backward Compatibility

- `ToolExecutionContext` alias maintains backward compatibility
- Existing code using `ToolExecutionContext` continues to work
- New code can use `BaseToolExecutionContext<TConfig>` for custom config types
- Default type parameter ensures minimal friction for consumers
